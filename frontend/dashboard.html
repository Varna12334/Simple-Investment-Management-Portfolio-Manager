// Fetch and render Predictive ML Insights async function loadPredictiveAnalytics() { try { const response = await fetch(`${API_BASE_URL}/analytics/forecast`); if (!response.ok) throw new Error("Could not retrieve forecasting analytics."); const data = await response.json(); const analyticsContainer = document.getElementById('ml-analytics-pane'); if (!analyticsContainer) return; analyticsContainer.innerHTML = ''; // Clear prior state if (data.forecasts.length === 0) { analyticsContainer.innerHTML = `

Add assets to activate predictive forecasting pipelines.

`; return; } data.forecasts.forEach(forecast => { const card = document.createElement('div'); card.className = "p-4 bg-slate-50 border border-slate-200 rounded-lg flex justify-between items-center mb-3"; card.innerHTML = `

${forecast['Asset Name']} (${forecast['Type']})

Expected 30D Target Range: $${forecast['Confidence Interval Lower']} - $${forecast['Confidence Interval Upper']}

$${forecast['Projected 30D Target']}
${forecast['Signal']}
`; analyticsContainer.appendChild(card); }); } catch (error) { console.error("ML Render Error:", error); } } // Append to your DOMContentLoaded pipeline inside your script: document.addEventListener('DOMContentLoaded', () => { // ... items from before loadPredictiveAnalytics(); });
