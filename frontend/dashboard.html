const API_BASE_URL = 'http://127.0.0.1:5000/api'; // Fetch metrics and holdings from the backend async function loadDashboardData() { try { const response = await fetch(`${API_BASE_URL}/portfolio`); if (!response.ok) throw new Error("Failed to fetch portfolio data."); const data = await response.json(); renderSummaryTotals(data.totals); renderHoldingsTable(data.assets); } catch (error) { console.error("Dashboard render error:", error); } } // Render the top scoreboard metrics function renderSummaryTotals(totals) { if (!totals) return; document.getElementById('total-invested').innerText = `$${(totals.total_invested || 0).toLocaleString(undefined, {minimumFractionDigits: 2})}`; document.getElementById('total-current').innerText = `$${(totals.total_current_value || 0).toLocaleString(undefined, {minimumFractionDigits: 2})}`; const pnlValue = totals.total_profit_loss || 0; const roiValue = totals.total_roi || 0; const pnlSpan = document.getElementById('total-pnl'); pnlSpan.innerText = `$${pnlValue.toLocaleString(undefined, {minimumFractionDigits: 2})} (${roiValue.toFixed(2)}%)`; // Dynamically handle accent coloring const pnlCard = pnlSpan.closest('.stat'); if (pnlCard) { if (pnlValue >= 0) { pnlCard.className = "stat profit"; } else { pnlCard.className = "stat loss"; } } } // Build table records dynamically function renderHoldingsTable(assets) { const tbody = document.getElementById('portfolio-table-body'); if (!tbody) return; tbody.innerHTML = ''; if (!assets || assets.length === 0) { tbody.innerHTML = `No assets trackable. Add your first trade entry below.`; return; } assets.forEach(asset => { const isProfit = asset['Profit/Loss'] >= 0; const row = document.createElement('tr'); row.innerHTML = ` ${asset['Asset Name']} ${asset['Type']} ${asset['Quantity']} $${parseFloat(asset['Buy Price']).toFixed(2)} $${parseFloat(asset['Current Price']).toFixed(2)} ${isProfit ? '+' : ''}$${parseFloat(asset['Profit/Loss']).toFixed(2)} (${parseFloat(asset['ROI (%)']).toFixed(2)}%) `; tbody.appendChild(row); }); } // Intercept Asset Entry Form Submissions async function submitNewAsset(event) { event.preventDefault(); const payload = { name: document.getElementById('name').value, type: document.getElementById('type').value, quantity: parseFloat(document.getElementById('quantity').value), buy_price: parseFloat(document.getElementById('buy_price').value), current_price: parseFloat(document.getElementById('current_price').value) }; try { const response = await fetch(`${API_BASE_URL}/invest`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); if (response.ok) { document.getElementById('asset-form').reset(); loadDashboardData(); // Refresh metrics instantly } else { alert("Error trying to persist item metrics."); } } catch (error) { console.error("Networking payload error:", error); } } // Bind initialization listeners document.addEventListener('DOMContentLoaded', () => { // Force route verification if (typeof checkSession === 'function') checkSession(); loadDashboardData(); const assetForm = document.getElementById('asset-form'); if (assetForm) { assetForm.addEventListener('submit', submitNewAsset); } });
